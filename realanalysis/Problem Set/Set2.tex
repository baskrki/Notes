\section{Problem Set 2}

\begin{problem}
    Let $a_n$ ad $b_n$ be a sequence of real numbers.
    \begin{enumerate}
        \item Assume that $\dps \lim_{n \to \infty} a_n$ and $\dps \lim_{n \to \infty} b_n$ exists. Show that \\ 
        $\dps \lim_{n \to \infty} (a_n b_n) = \left( \dps \lim_{n \to \infty} a_n \right)
        \left( \dps \lim_{n \to \infty} a_n \right)$ exists.
        \item Give an example in which $\dps \lim_{n \to \infty} (a_n b_n)$ exists but neither 
        $\dps \lim_{n \to \infty} a_n$ nor $\dps \lim_{n \to \infty} b_n$ exits.
    \end{enumerate} 
\end{problem}

\textit{Solutions.}

To show that $\dps \lim_{n \to \infty} a_n b_n= ab$ we need to show $\exists N \in \NN$ such that for all $\eps > 0$ we have
\[ |a_nb_n-ab| < \eps \]
for $n \ge N$. Now,
\begin{align*}
    |a_nb_n - ab | &= |a_nb_n - a_n b + a_n b - ab| \\
    &\le |a_n(b_n-b)| + |b(a_n-a)|\\
    &= |a_n||b_n-b| + |b| |a_n-a|
\end{align*}
Since $\set{a_n}$ is convergent, we know that it is bounded above. Suppose $|a_n| < C$ then
\[ |a_n||b_n-b| + |b| |a_n-a| < |C| |b_n-b| + |b||a_n-a| \]
Also let $N_1 \in \NN$ such that $\dps |b_n-b| < \frac{\eps}{2(|C|+1)}$ for all $n \ge N_1$
and let $N_2 \in \NN$ such that $\dps |a_n - a| < \frac{\eps}{2(|b|+1)}$ for all $n \ge N_2$. Thus for $n \ge \max\{N_1,N_2\}$ we have,
\begin{align*}2
    |a_n||b_n-b| + |b| |a_n-a| &< |C| |b_n-b| + |b||a_n-a| \\
    &< |C| \cdot \frac{\eps}{2(|C|+1)} + |b| \cdot \frac{\eps}{2(|b|+1)} \\
    &< \frac{\eps}{2} + \frac{\eps}{2} \\
    &< \eps 
\end{align*}

\vspace{6pt}
For the second part just take $a_n=(-1)^n$ and $b_n=(-1)^n$. We know that $\dps \lim_{n \to \infty} a_n b_n =1$ but 
$\dps \lim_{n \to \infty} a_n$ and $\dps \lim_{n \to \infty} b_n$ doesn't exist.

\begin{problem}
    Find the limit of the following sequence if it exists or show that the limit doesn't exists.
    \begin{enumerate}
        \item $a_n = \frac{n^2}{n+1} - \frac{n^2+1}{n}$
        \item $a_n = \frac{\sin(n)}{n}$
        \item $\dps a_n = \sum_{i=1}^{n} \frac{n^2}{\sqrt{n^6+i}}$
    \end{enumerate}
\end{problem}

\textit{Solution.}

For the first part we know that $a_n = \frac{1}{n+1}-\frac{1}{n}-1$.
Thus,
\begin{align*}
    \lim_{n \to \infty} \frac{1}{n+1}-\frac{1}{n}-1 &= \lim_{n \to \infty} \frac{1}{n+1} - \lim_{n \to \infty} \frac{1}{n} - \lim_{n \to \infty} 1 \\
    &= 0-0-1 \\
    &=-1
\end{align*}
The first equation is due to the algebraic property of of limit.

\vspace{6pt}
For the second part, we need to show that
\[ \abs{\frac{\sin (n)}{n}} < \eps \]
We know that $0 \le |\sin(n)| \le 1 \Rightarrow 0 \le \frac{|\sin(n)|}{|n|} \le \frac{1}{|n|}$.
Thus, for any $N > \frac{1}{\eps}$ we have 
\[ \abs{\frac{\sin(n)}{n}} \le \frac{1}{|n|} < \frac{1}{N} < \eps \]
